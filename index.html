1.5 Forms

Common styles for form elements

Normal 

<form role="form">
    <fieldset>
        <p>
            <label for="exampleInputEmail1">Email</label>
            <input type="email" id="exampleInputEmail1" placeholder="Email">
        </p>

        <p>
            <label for="exampleInputPassword1">Password</label>
            <input type="password" id="exampleInputPassword1" placeholder="Password">
        </p>

        <p>
            <label>
                Choose State <span class="o-tag t-form-info">This is a required field.</span>
            </label>
            <select>
                <option>AL</option>
                <option>CA</option>
                <option>IL</option>
                <option>MD</option>
                <option>NY</option>
            </select>
        </p>

        <p>
            <input type="checkbox" id="checkbox1" />
            <label for="checkbox1">Check me out</label>
        </p>

        <p>
            <button type="submit" class="o-btn">Submit</button>
        </p>
    </fieldset>
</form>
Inline 
<form role="form">
    <fieldset>
        <span>
            <label class="sr-only" for="exampleInputEmail2">Email</label>
            <input type="email" id="exampleInputEmail2" placeholder="Email">
        </span>

        <span>
            <label class="sr-only" for="exampleInputPassword2">Password</label>
            <input type="password" id="exampleInputPassword2" placeholder="Password">
        </span>

        <span>
            <input type="checkbox" id="checkbox2" />
            <label for="checkbox2">Remember me</label>
        </span>

        <button type="submit" class="o-btn">Submit</button>
    </fieldset>
</form>
